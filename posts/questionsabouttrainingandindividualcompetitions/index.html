Problem Description
Fibonacci numbers are well-known as follow:

f(x)=1  x=1
f(x)=2  x=2
f(x)=f(x-1)+f(x-2)  x>2

Input
Multiple test cases, the first line is an integer T (T<=10000), indicating the number of test cases.

Each test case is a line with an integer N (1<=N<=1e9).

One line per case. If the answer don’t exist, output “-1” (without quotes). Otherwise, your answer should be formatted as “N=f1+f2+…+fn”. N indicates the given number and f1, f2, … , fn indicating the Fibonacci numbers in ascending order. If there are multiple ways, you can output any of them in the order of smallest to largest .

//小于等于n的最大的斐波那契数
int fib(int n)
{
    if (n == 1 or n == 2)
        return n;
    int a = 1, b = 2, c = 3;
    while (c <= n)
    {
        int t = b;
        c += b;
        b += a;
        a = t;
    }
    return b;
}

Problem: Swiss-system tournament

Problem Description
A Swiss-system tournament is a tournament which uses a non-elimination format. The first tournament of this type was a chess tournament in Zurich in 1895, hence the name "Swiss system". The tournament will be held based on following rules.

2*N contestants (indexed 1, 2, ..., 2*N) will have R rounds matches. Before the first round, every contestant has an origin score. After every match, winner will get 1 score and loser will get 0 score. Before and after every round, contestants will be sorted by their scores in descending order. Two contestants with the same score will be sorted by their index with ascending order.

In every round, contestants will have match based on the sorted list. The first place versus the second place, the third place versus the forth place, ..., the Kth place versus the (K + 1)th place, ..., the (2*N - 1)th place versus (2*N)th place.

Now given the origin score and the ability of every contestant, we want to know the index of the Qth place contestant. We ensured that there won’t be two contestants with the same ability and the contestant with higher ability will always win the match.

Input
Multiple test cases. The first line contains a positive integer T (T<=10) indicating the number of test cases.

For each test case, the first line contains three positive integers N (N <= 100,000), R (R <= 50), Q (Q <= 2*N), separated by space.

The second line contains 2*N non-negative integers, s1, s2, ..., s2*N, si (si<= 10^8) indicates the origin score of constant indexed i.

The third line contains 2*N positive integers, a1, a2, ..., a2*N, ai(ai<= 10^8) indicates the ability of constant indexed i.

Output
One line per case, an integer indicates the index of the Qth place contestant after R round matches.

Sample Input
1
2 4 2
7 6 6 7

1

using namespace std;
const int N = 1e5 + 10;
struct contestant
{
    int index, score, alility;
} all[2 * N], win[N], lose[N];
int n, r, q;

bool cmp(contestant& a, contestant& b)
{
    if (a.score == b.score)
        return a.index < b.index;
    else
        return a.score > b.score;
}

void merge_sort()
{
    int cntwin = 0, cntlose = 0;
    for (int i = 0; i < 2 * n; i += 2)
    {
        if (all[i].alility > all[i + 1].alility)
        {
            all[i].score++;
            win[cntwin++]   = all[i];
            lose[cntlose++] = all[i + 1];
        }
        else
        {
            all[i + 1].score++;
            win[cntwin++]   = all[i + 1];
            lose[cntlose++] = all[i];
        }
    }
    merge(win, win + cntwin, lose, lose + cntlose, all, cmp);
}

int main(void)
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int t;
    cin >> t;
    while (t--)
    {
        cin >> n >> r >> q;
        for (int i = 0; i < 2 * n; i++)
        {
            all[i].index = i + 1;
            cin >> all[i].score;
        }
        for (int i = 0; i < 2 * n; i++)
        {
            cin >> all[i].alility;
        }
        sort(all, all + 2 * n, cmp);
        while (r--)
        {
            merge_sort();
        }
        cout << all[q - 1].index << endl;
    }
    return 0;
}

int fa[maxn];
int main(void)
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n, m, t;
    cin >> n >> m >> t;
    for (int i = 0; i < m; i++)
    {
        int a, b;
        cin >> a >> b;
        edge[b].push_back(a); // 反向建边
        if (b == t)
            p.insert(a); //记录和t直接连接的点
        // else
        //     edge[b].push_back(a);   //对应if(now == t)
    }
    queue<int> q;
    for (auto&& i : p)
    {
        q.push(i);
        cnt[i]++;  //入队cnt++,但是它不能被自己所连通的点(不在p中)所更新,也是为了避免成环,所以就有了fa[]数组
        fa[i] = i; //p中每个点 fa[] 等于它自己, 它所能访问到的点的fa[]也等于它
        while (!q.empty())
        {
            int now = q.front();
            q.pop();
            if (now == t) //不要处理t点,因为p中是t直接相连的点;        或者当初t点相连的就不建边,此判断就可省略
                continue;
            for (auto&& j : edge[now])
            {
                if (cnt[j] >= 2 or fa[j] == fa[now])
                    continue;
                cnt[j]++, fa[j] = fa[now], q.push(j);
            }
        }
    }
    vector<int> ans;
    for (auto&& i : p)
    {
        if (cnt[i] == 1)
            ans.push_back(i);
    }
    cout << ans.size() << endl;
    for (auto&& i : ans)
    {
        cout << i << endl;
    }
    return 0;
}

struct queneNode
{
    int id;
    int distance;
    int co2;
    queneNode(int a, int b, int c)
        : id(a), distance(b), co2(c)
    {
    }
    bool operator<(const queneNode& n) const
    {
        if (co2 == n.co2)
            return distance > n.distance;
        return co2 > n.co2;
    }
};
int dist[1005][105]; //二维迪杰斯特拉算法,走到第i个顶点,公里数为j时的最优解
bool vis[1005][105]; //vis也要对应变成二维的

void dijkstra()
{
    memset(dist, 0x3f, sizeof(dist));
    priority_queue<queneNode> q;
    q.push(queneNode(1001, 0, 0));
    while (!q.empty())
    {
        queneNode now = q.top();
        q.pop();
        if (vis[now.id][now.distance])
            continue;
        vis[now.id][now.distance] = true;
        for (auto&& i : edge[now.id])
        {
            if (now.distance + i.distance <= maxDistance and !vis[i.to][now.distance + i.distance] and i.co2 + now.co2 < dist[i.to][now.distance + i.distance])
            {
                dist[i.to][now.distance + i.distance] = i.co2 + now.co2;
                q.push(queneNode(i.to, now.distance + i.distance, dist[i.to][now.distance + i.distance]));
            }
        }
    }
}
