int qmul(int a, int b, int mod)
{
    long long ans = 0;
    while (b)
    {
        if (b & 1)
            ans = (ans + a) % mod;
        a = (a + a) % mod;
        b >>= 1;
    }
    return ans;
}

乘法逆元

https://www.cnblogs.com/mjtcn/p/7241896.html

对于个人的理解逆元，逆元是在运算除时，可以变成乘，方便计算，像上面一样，a/b(mod p)就可以变成a*b-1，这也就是逆元的本质，必须要在模的意义下才有效。

运算a/b时，我们除以b就相当于乘以1/b，这是个分数，不利于计算，所以我们就找到了一个整数，b的逆元，比如计算12/3（mod 7），这是个除法，所以我们可以这样12*(1/3)，虽然是乘法了，但是有分数，所以找一个整数x，使得 3x≡1(mod 7)，x=5，即模 7意义下3的逆元是5，然后我们乘以这个逆元，12*5 = 60,60 mod 7 = 4，诶，12/3 = 4，相等啊，这就是上面所说的性质

如果模数是质数,可以利用费马小定理,求逆元

$p是质数,a是任意自然数,则a^p\equiv a \mod p$ 此为费马小定理

$即 a^{p-1} \equiv 1 \mod p $

由$ax\equiv 1 \mod p$

-> $ ax \equiv a^{p-1} \mod p$

-> $x \equiv a ^{p-2} \mod p$

void exgcd(int a, int b, int& x, int& y)
{
    if (b == 0)
    {
        x = 1, y = 0;
        return;
    }
    exgcd(b, a % b, y, x);
    y -= a / b * x;
}

int inv(int a, int mod)
{
    //要求gcd(a,mod) == 1
    int x, y;
    exgcd(a, mod, x, y);
    return (x % mod + mod) % mod;
}

int fact(int n, int mod)
{
    long long ans = 1;
    for (int i = 2; i <= n; i++)
        ans = ans * i % mod;
    return ans;
}

int C(int n, int m, int mod)
{
    if (n < 0 || m < 0 || m > n)
        return 0;
    if (n == m)
        return 1;
    return (fact(n, mod) % mod * inv(fact(m, mod), mod) % mod * inv(fact(n - m, mod), mod) % mod) % mod;
}

//Lucas 定理,当n和m较大,明显优化
int Lucas(int n, int m, int p)
{
    if (m == 0)
        return 1;
    return C(n % p, m % p, p) * Lucas(n / p, m / p, p) % p;
}

欧拉函数

  • 欧拉函数写做$\varphi $，表示小于n的正整数(1 <= n )中与n互质的数的数目

  • $\varphi[1] = 1 $

  • $若p是质数, \varphi[p] = p -1$

  • 欧拉函数是积性函数

  • $gcd(a,b) = 1 \rightarrow \varphi[a\times b] = \varphi[a]\times \varphi[b]$

  • 当b是奇数,$\varphi[2*b] = \varphi[b] \times \varphi[2] = \varphi[b]$

  • 若 $n = p^k$ ，其中 $p$ 是质数，那么 $\varphi(n) = p^k - p^{k - 1}$

    • 证明：显然对于从 1 到 $p^k$ 的所有数中，除了 $p^{k-1}$ 个 $p$ 的倍数以外其它数都与 $p^k$ 互素，故 $\varphi(p^k)=p^k-p^{k-1}=p^{k-1}\times(p-1)$ ，证毕。

  • $n中与n互质的数的和为\phi(n)/2*n(n>1)$

  • $\phi(n)(n>2)为偶数$

    • 因为与n互质的数都是成对出现的，所以显然与n互质的数为偶数，即ϕ(n)为偶数。且每一对的和都为n .

  • 设 $n = \prod_{i=1}^{n}p_i^{k_i}$ ，其中 $p_i$ 是质因子，有 $\varphi(n) = n \times \prod_{i = 1}^s{\dfrac{p_i - 1}{p_i}}$

  • $$ \begin{aligned} \varphi(n) &= \prod_{i=1}^{s} \varphi(p_i^{k_i}) \ &= \prod_{i=1}^{s} (p_i-1)\times {p_i}^{k_i-1}\ &=\prod_{i=1}^{s} {p_i}^{k_i} \times(1 - \frac{1}{p_i})\ &=n~ \prod_{i=1}^{s} (1- \frac{1}{p_i}) \end{aligned} $$

  • $p是质数,\left{ \begin{aligned} \varphi[p]=p-1\ ————–①\ \varphi[ip]=p\varphi[i]\ \ \ \ \ \ \ \ \ \ \ \ (i\bmod p==0)—②\ \varphi[ip]=(p-1)\varphi[i]\ \ \ \ \ (i\bmod p\ne0)—③ \end{aligned} \right.$

  • 设 $p_1$ 是 $n$ 的最小质因子， $n' = \frac{n}{p_1}$ ，那么线性筛的过程中 $n$ 通过 $n' \times p_1$ 筛掉。

  • 如果 $n' \bmod p_1 = 0$ ，那么 $n'$ 包含了 $n$ 的所有质因子

    • $$ \begin{aligned} \varphi(n) & = n \times \prod_{i = 1}^s{\frac{p_i - 1}{p_i}} \ & = p_1 \times n' \times \prod_{i = 1}^s{\frac{p_i - 1}{p_i}} \ & = p_1 \times \varphi(n') \end{aligned} $$

  • 那如果 $n' \bmod p_1 \neq 0$ 呢，这时 $n'$ 和 $p_1$ 是互质的，根据欧拉函数性质，我们有：

    $$ \begin{aligned} \varphi(n) & = \varphi(p_1) \times \varphi(n') \ & = (p_1 - 1) \times \varphi(n') \end{aligned} $$

int euler_phi(int n)
{
    int ans = n;
    for (int i = 2; i * i <= n; i++)
    {
        if (n % i == 0)
        {                            //i是n的质因子
            ans = ans / i * (i - 1); //先除后乘，防止溢出
            while (n % i == 0)
                n /= i;
        }
    }
    if (n > 1) //处理到n为质数
        ans = ans / n * (n - 1);
    return ans;
}

const int maxn = 1e6;
bool isnp[maxn];   //is not prime , 这个数组可以省略
int phi[maxn];     // 欧拉函数表
vector<int> prime; //质数表
void phi_table(int n)
{
    memset(isnp, 0, sizeof(isnp));
    phi[1] = 1;
    for (int i = 2; i <= n; i++)
    {
        if (!isnp[i])
        {
            prime.push_back(i);
            phi[i] = i - 1; //phi[p] = p - 1
        }
        for (int j : prime)
        {
            if (1ll * i * j > n)
                break;
            isnp[i * j] = true;
            if (i % j == 0) //n = j * i , j是n的最小质因子 , i 包含 n 的所有质因子
            {
                phi[i * j] = j * phi[i]; //phi(n) = n * ... = j * (i * ...) = j * phi(i)
                break;
            }
            else
            {
                phi[i * j] = (j - 1) * phi[i]; //gcd(i,j) = 1 , phi(i*j) = phi(j) * phi(i) = (p - 1) * phi(i)
            }
        }
    }
}

欧拉定理/费马小定理

  • 费马小定理

    • 若 $p$ 为素数， $\gcd(a, p) = 1$ ，则 $a^{p - 1} \equiv 1 \pmod{p}$ 。另一个形式：对于任意整数 $a$ ，有 $a^p \equiv a \pmod{p}$ 。

  • 欧拉定理

  • 若 $\gcd(a, m) = 1$ ，则 $a^{\varphi(m)} \equiv 1 \pmod{m}$ 。

  • 欧拉定理是费马小定理的推广

  • 扩展欧拉定理

    • $a^b\equiv \begin{cases} a^{b%\phi(p)}gcd(a,p)=1\ a^bb<\phi(p)\ a^{b%\phi(p)+\phi(p)}~~~~~b\geq\phi(p) \end{cases}(mod~p)$

int euler_phi(int n)
{
    int m   = (int)sqrt(n + 0.5);
    int ans = n;
    for (int i = 2; i <= m; i++)
    {
        if (n % i == 0)
        {
            ans = ans / i * (i - 1);
            while (n % i == 0)
                n /= i; //除尽
        }
    }
    if (n > 1)
        ans = ans / n * (n - 1); //剩下的不为1,也是素数
    return ans;
}

int qpow(int a, int b, int p)
{
    int ret = 1;
    while (b)
    {
        if (b & 1)
            ret = Mod(ret * a, p);
        a = Mod(a * a, p);
        b >>= 1;
    }
    return ret;
}

#include <bits/stdc++.h>

using namespace std;
#define debug(x) std::cout << #x << " is " << (x) << endl;
#define int long long

const int MAXN = 1.6e8 / 2 + 1000;
const int mod  = 1e9 + 7;

bitset<MAXN> vis;
vector<int> prime;

void euler()
{
    for (int i = 2; i < MAXN; i++)
    {
        if (!vis[i])
        {
            prime.push_back(i);
        }
        for (auto&& j : prime)
        {
            if (i * j > MAXN)
                break;
            vis[i * j] = true;
            if (i % j == 0)
                break;
        }
    }
}

int qpow(int a, int b)
{
    int ans = 1;
    while (b)
    {
        if (b & 1)
        {
            ans *= a;
            ans %= mod;
        }
        a *= a;
        a %= mod;
        b >>= 1;
    }
    return ans;
}

signed main(void)
{
    euler();
    int n;
    cin >> n;
    int ans = 1;
    for (auto&& i : prime)
    {
        if (i > n / 2)
            break;
        int k;
        if (i == 2)
            k = log2(n / 3);
        else
            k = log(n / 2) / log(i);
        ans = (ans * qpow(i, k)) % mod;
    }
    if (ans == 1)
        cout << "empty";
    else
        cout << ans;
    return 0;
}

using namespace std;
#define debug(x) std::cout << #x << " is " << (x) << endl;
#define int long long
const int MAXN = 1e5 + 100;
vector<int> edge[MAXN];

int id[MAXN] = {1}, cnt = 0; //注意id[0] = 1
void dfs(int now, int fa)    //从叶子节点开始分组
{
    for (int i : edge[now])
    {
        if (i != fa)
            dfs(i, now);
    }
    if (!id[now])
    {
        if (id[fa])
        {
            cout << -1;
            exit(0);
        }
        id[fa] = id[now] = ++cnt;
    }
}

char color[MAXN];
void dfs2(int now, int fa, bool state) //从根节点染色
{
    color[now] = state ? 'R' : 'B';
    for (int i : edge[now])
    {
        if (i != fa)
            dfs2(i, now, id[i] == id[now] ? state : !state); //有且仅有一个相同颜色
    }
}

signed main(void)
{
    int n;
    cin >> n;
    for (int i = 0; i < n - 1; i++)
    {
        int a, b;
        cin >> a >> b;
        edge[a].push_back(b);
        edge[b].push_back(a);
    }
    dfs(1, 0);
    dfs2(1, 0, 0);
    for (int i = 1; i <= n; i++)
    {
        cout << color[i];
    }
    return 0;
}

#include <bits/stdc++.h>

using namespace std;
#define debug(x) std::cout << #x << " is " << (x) << endl;
#define int long long

const int mod = 1e9 + 7;

int qpow(int a, int b)
{
    int res = 1;
    while (b)
    {
        if (b & 1)
        {
            res = res * a % mod;
        }
        a = a * a % mod;
        b >>= 1;
    }
    return res;
}

int inv(int a)
{
    return qpow(a, mod - 2);
}

int fa[510 * 510];
int num[510 * 510];

int n;
unsigned id(int a, int b)
{
    return a * n + b;
}

void init()
{
    for (int i = 0; i < 510 * 510; i++)
    {
        fa[i]  = i;
        num[i] = 1;
    }
}

unsigned _find(int a)
{
    return fa[a] == a ? a : fa[a] = _find(fa[a]);
}

void _union(int a, int b)
{
    a = _find(a);
    b = _find(b);
    if (a != b)
    {
        fa[b] = a;
        num[a] += num[b];
    }
}

char board[510][510];
int dir[4][2] = {1, 0, -1, 0, 0, 1, 0, -1};

bool islegal(int x, int y)
{
    return x >= 0 and x < n and y >= 0 and y < n;
}

signed main(void)
{
    cin >> n;
    init();
    for (int i = 0; i < n; i++)
    {
        cin >> board[i];
    }
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < n; j++)
        {
            if (board[i][j] == '1')
            {
                for (int k = 0; k < 4; k++)
                {
                    int x = i + dir[k][0];
                    int y = j + dir[k][1];
                    if (islegal(x, y) and board[x][y] == '1')
                    {
                        _union(id(x, y), id(i, j));
                    }
                }
            }
        }
    }
    int ans = 1, cnt = 0;
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < n; j++)
        {
            if (board[i][j] == '1' and fa[id(i, j)] == id(i, j))
            {
                cnt++;
                ans = ans * num[id(i, j)] % mod;
            }
        }
    }
    for (int i = 1; i <= cnt; i++)
    {
        ans = ans * i % mod;
    }
    int _n;
    cin >> _n;
    while (_n--)
    {
        int a, b;
        cin >> a >> b;
        if (board[a][b] == '1')
        {
            cout << ans << endl;
            continue;
        }
        board[a][b] = '1';
        cnt++;
        ans = ans * cnt % mod;
        for (int k = 0; k < 4; k++)
        {
            int x = a + dir[k][0];
            int y = b + dir[k][1];
            if (islegal(x, y) and board[x][y] == '1')
            {
                int f1 = _find(id(a, b));
                int f2 = _find(id(x, y));
                if (f1 != f2)
                {
                    ans = ans * inv(cnt) % mod;
                    ans = ans * inv(num[f1]) % mod;
                    ans = ans * inv(num[f2]) % mod;
                    ans = ans * (num[f1] + num[f2]) % mod;
                    cnt--;
                    _union(f1, f2);
                }
            }
        }
        cout << ans << endl;
    }
    return 0;
}

const int maxn = 1e5 + 1000;
typedef long long LL;
inline LL read()
{
    LL x = 0, f = 1;
    char ch = getchar();
    while (!isdigit(ch))
    {
        if (ch == '-')
            f = -1;
        ch = getchar();
    }
    while (isdigit(ch))
    {
        x  = x * 10 + ch - 48;
        ch = getchar();
    }
    return x * f;
}
LL phi[maxn], primes[maxn], cnt;
bool v[maxn];
void Euler(LL num)
{
    for (LL i = 2; i < num; i++)
    {
        if (!v[i])
        {
            primes[++cnt] = i;
            phi[i]        = i - 1;
        }
        for (LL j = 1; j <= cnt; j++)
        {
            if (primes[j] * i > num)
                break;
            v[primes[j] * i] = 1;
            if (i % primes[j] == 0)
            {
                phi[i * primes[j]] = phi[i] * primes[j];
                break;
            }
            phi[i * primes[j]] = phi[i] * (primes[j] - 1);
        }
    }
    phi[1] = 1;
}
LL q_pow(LL a, LL b, LL p)
{
    LL res = 1;
    while (b)
    {
        if (b & 1)
            res = res * a > p ? res * a % p + p : res * a; //mod(x,p) = x < p ? x : x%p + p;
        b >>= 1;
        a = a * a > p ? a * a % p + p : a * a;
    }
    return res;
}
LL solve(LL a, LL b, LL m)
{
    if (m == 1 || b == 0)
        return 1;
    return q_pow(a, solve(a, b - 1, phi[m]), m);
}
